357  template<typename... Locks>
358  static void exec_tx(pool_base &pool, std::function<void ()> tx,
359  Locks &... locks)
360  {
361  if (pmemobj_tx_begin(pool.get_handle(),
362  NULL, TX_LOCK_NONE) != 0)
363  throw transaction_error(
364  "failed to start transaction");
365 
366  auto ret = add_lock(locks...);
367  if (ret) {
368  pmemobj_tx_abort(ret);
369  pmemobj_tx_end();
370  throw transaction_error(
371  "failed to add a lock to the"
372  " transaction");
373  }
374 
375  try {
376  tx();
377  } catch (manual_tx_abort &) {
378  pmemobj_tx_end();
379  throw;
380  } catch (...) {
381  /* first exception caught */
382  if (pmemobj_tx_stage() == TX_STAGE_WORK) {
383  pmemobj_tx_abort(ECANCELED);
384  }
385  /* waterfall tx_end for outer tx */
386  pmemobj_tx_end();
387  throw;
388  }
389 
390  auto stage = pmemobj_tx_stage();
391  if (stage == TX_STAGE_WORK) {
392  pmemobj_tx_commit();
393  } else if (stage == TX_STAGE_ONABORT) {
394  pmemobj_tx_end();
395  throw transaction_error("transaction aborted");
396  } else if (stage == TX_STAGE_NONE)
397  throw transaction_error("transaction ended"
398  "prematurely");
399 
400  pmemobj_tx_end();
401  }
